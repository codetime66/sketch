
#1 - criar o diretorio para o bundle
mkdir /meucontainer
cd /meucontainer


#2 - criar o subdiretorio rootfs 
mkdir rootfs


#3 - exportar o busybox via Docker dentro do subdiretorio rootfs
docker export $(docker create busybox) | tar -C rootfs -xvf -


#4 - criar o arquivo de configuracao OCI
runc spec


#5 - executar o container
cd /mycontainer
runc run mycontainerid
